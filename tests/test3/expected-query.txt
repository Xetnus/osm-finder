WITH intersections AS
(
  SELECT
    ((ST_DumpPoints(
      ST_Intersection(line1.geom, line2.geom)
    )).geom) AS intersection1,
    ((ST_DumpPoints(
      ST_Intersection(line1.geom, line3.geom)
    )).geom) AS intersection2,
    line1.geom AS line1_geom,
    line1.osm_id AS line1_id,
    line1.osm_type AS line1_type,
    line2.geom AS line2_geom,
    line2.osm_id AS line2_id,
    line2.osm_type AS line2_type,
    line3.geom AS line3_geom,
    line3.osm_id AS line3_id,
    line3.osm_type AS line3_type
  FROM linestrings AS line1, linestrings AS line2, linestrings AS line3
  WHERE line1.category = 'roadway' AND line2.category = 'roadway' AND line2.tags->>'bridge' = 'yes' AND line3.category = 'roadway' AND line3.tags->>'bridge' = 'yes' AND line3.osm_id != line2.osm_id AND line3.osm_id != line1.osm_id AND line2.osm_id != line1.osm_id AND ST_Intersects(line1.geom, line2.geom) AND ST_Intersects(line1.geom, line3.geom) AND ST_DWithin(line2.geom, line3.geom, 200) 
),
buffers AS
(
  SELECT
    intersections.intersection1,
    ST_ExteriorRing(ST_Buffer(intersections.intersection1, 0.5)) AS ring1,
    intersections.intersection2,
    ST_ExteriorRing(ST_Buffer(intersections.intersection2, 0.5)) AS ring2,
    intersections.line1_geom,
    intersections.line1_id,
    intersections.line1_type,
    intersections.line2_geom,
    intersections.line2_id,
    intersections.line2_type,
    intersections.line3_geom,
    intersections.line3_id,
    intersections.line3_type
  FROM intersections
),
points AS
(
  SELECT
    ST_GeometryN
    (
      ST_Intersection(buffers.ring1, buffers.line1_geom)
      , 1
    ) AS ring1_p1,
    ST_GeometryN
    (
      ST_Intersection(buffers.ring1, buffers.line2_geom)
      , 1
    ) AS ring1_p2,
    ST_GeometryN
    (
      ST_Intersection(buffers.ring2, buffers.line1_geom)
      , 1
    ) AS ring2_p1,
    ST_GeometryN
    (
      ST_Intersection(buffers.ring2, buffers.line3_geom)
      , 1
    ) AS ring2_p2,
    buffers.intersection1,
    buffers.ring1,
    buffers.intersection2,
    buffers.ring2,
    buffers.line1_geom,
    buffers.line1_id,
    buffers.line1_type,
    buffers.line2_geom,
    buffers.line2_id,
    buffers.line2_type,
    buffers.line3_geom,
    buffers.line3_id,
    buffers.line3_type
  FROM buffers
)
SELECT
  points.line1_type AS line1_type,
  points.line1_id AS line1_id,
  points.line2_type AS line2_type,
  points.line2_id AS line2_id,
  points.line3_type AS line3_type,
  points.line3_id AS line3_id
FROM points
WHERE
(
  (
    abs(round(degrees(
      ST_Azimuth(points.ring1_p2, points.intersection1)
      -
      ST_Azimuth(points.ring1_p1, points.intersection1)
    ))::decimal % 180.0) BETWEEN 88 AND 92
  ) 
)
AND 
(
  (
    abs(round(degrees(
      ST_Azimuth(points.ring2_p2, points.intersection2)
      -
      ST_Azimuth(points.ring2_p1, points.intersection2)
    ))::decimal % 180.0) BETWEEN 55 AND 75
  ) 
  OR
  (
    abs(round(degrees(
      ST_Azimuth(points.ring2_p2, points.intersection2)
      -
      ST_Azimuth(points.ring2_p1, points.intersection2)
    ))::decimal % 180.0) BETWEEN 105 AND 125
  ) 
);
