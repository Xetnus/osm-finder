SELECT
  line1.osm_id AS line1_id, line1.osm_type AS line1_type, 
  line2.osm_id AS line2_id, line2.osm_type AS line2_type
FROM linestrings AS line1, linestrings AS line2
WHERE line1.category = 'railway' AND line1.subcategory = 'abandoned' AND line2.category = 'railway' AND line2.subcategory = 'abandoned' AND line2.tags->>'bridge' = 'yes' AND line2.osm_id != line1.osm_id AND ST_DWithin(line1.geom, line2.geom, 100) AND 
(
  (
    abs((
      degrees(ST_Azimuth(ST_StartPoint(line2.geom), ST_EndPoint(line2.geom)))
      -
      degrees(ST_Azimuth(ST_StartPoint(line1.geom), ST_EndPoint(line1.geom)))
    )::decimal % 180.0) BETWEEN 155 AND 175
  ) 
  OR
  (
    abs((
      degrees(ST_Azimuth(ST_StartPoint(line2.geom), ST_EndPoint(line2.geom)))
      -
      degrees(ST_Azimuth(ST_StartPoint(line1.geom), ST_EndPoint(line1.geom)))
    )::decimal % 180.0) BETWEEN 5 AND 25
  ) 
)
;
